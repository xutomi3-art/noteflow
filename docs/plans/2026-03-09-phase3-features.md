PPT_PROMPT = """You are creating a PowerPoint presentation based on the source documents.
Generate a JSON structure for a presentation with the following format:
{
  "title": "Presentation Title",
  "slides": [
    {
      "title": "Slide Title",
      "bullets": ["Point 1", "Point 2", "Point 3"]
    }
  ]
}

Rules:
1. Generate 5-8 slides maximum
2. Each slide has 3-5 bullet points
3. Keep bullets concise (under 15 words each)
4. First slide is a title/overview slide
5. Return ONLY valid JSON, no markdown fences

Source content:
{context}"""


@router.post("/{notebook_id}/studio/ppt")
async def generate_ppt(
    notebook_id: uuid.UUID,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user),
):
    await permission_service.check_permission(db, notebook_id, current_user.id, "viewer")
    context = await _get_source_context(db, notebook_id)
    if not context:
        raise HTTPException(status_code=400, detail="No source documents available")

    prompt = PPT_PROMPT.format(context=context[:8000])
    raw = await qwen_client.generate(
        [{"role": "system", "content": "Return only valid JSON."},
         {"role": "user", "content": prompt}]
    )

    # Strip markdown fences if present
    raw = raw.strip()
    if raw.startswith("```"):
        raw = raw.split("\n", 1)[1].rsplit("```", 1)[0].strip()

    try:
        data = json_lib.loads(raw)
    except Exception:
        raise HTTPException(status_code=500, detail="Failed to parse slide structure from AI")

    # Build .pptx in memory
    prs = Presentation()
    prs.slide_width = Inches(13.33)
    prs.slide_height = Inches(7.5)
    blank_layout = prs.slide_layouts[6]  # blank layout

    # Title slide
    title_slide = prs.slides.add_slide(prs.slide_layouts[0])
    title_slide.shapes.title.text = data.get("title", "Presentation")
    title_slide.placeholders[1].text = "Generated by Noteflow AI"

    # Content slides
    content_layout = prs.slide_layouts[1]  # title + content
    for slide_data in data.get("slides", []):
        slide = prs.slides.add_slide(content_layout)
        slide.shapes.title.text = slide_data.get("title", "")
        tf = slide.placeholders[1].text_frame
        tf.clear()
        bullets = slide_data.get("bullets", [])
        for i, bullet in enumerate(bullets):
            if i == 0:
                tf.paragraphs[0].text = bullet
            else:
                p = tf.add_paragraph()
                p.text = bullet
                p.level = 0

    buf = io.BytesIO()
    prs.save(buf)
    buf.seek(0)

    filename = f"{data.get('title', 'presentation').replace(' ', '_')}.pptx"
    return StreamingResponse(
        buf,
        media_type="application/vnd.openxmlformats-officedocument.presentationml.presentation",
        headers={"Content-Disposition": f'attachment; filename="{filename}"'}
    )
```

async downloadPPT(notebookId: string): Promise<void> {
  const response = await fetch(`/api/notebooks/${notebookId}/studio/ppt`, {
    method: 'POST',
    credentials: 'include',
  });
  if (!response.ok) {
    const err = await response.json();
    throw new Error(err.detail || 'PPT generation failed');
  }
  const blob = await response.blob();
  const url = URL.createObjectURL(blob);
  const a = document.createElement('a');
  a.href = url;
  a.download = response.headers.get('content-disposition')
    ?.match(/filename="(.+)"/)?.[1] || 'presentation.pptx';
  a.click();
  URL.revokeObjectURL(url);
},

'use client';
import React, { useMemo } from 'react';
import ReactFlow, {
  Node, Edge, Background, Controls,
  useNodesState, useEdgesState,
} from 'reactflow';
import 'reactflow/dist/style.css';

interface MindMapData {
  nodes: Array<{ id: string; label: string; level: number; parent?: string }>;
}

function buildGraph(data: MindMapData): { nodes: Node[]; edges: Edge[] } {
  const levelColors = ['#3b82f6', '#8b5cf6', '#10b981'];
  const levelOffsets: Record<number, number> = { 0: 0, 1: 0, 2: 0 };

  const nodes: Node[] = data.nodes.map((n) => {
    const x = n.level * 300;
    const y = (levelOffsets[n.level] ?? 0) * 120;
    levelOffsets[n.level] = (levelOffsets[n.level] ?? 0) + 1;
    return {
      id: n.id,
      position: { x, y },
      data: { label: n.label },
      style: {
        background: levelColors[n.level] ?? '#6b7280',
        color: '#fff',
        border: 'none',
        borderRadius: '8px',
        padding: '8px 16px',
        fontSize: n.level === 0 ? '14px' : '12px',
        fontWeight: n.level === 0 ? 600 : 400,
        minWidth: '120px',
        textAlign: 'center',
      },
    };
  });

  const edges: Edge[] = data.nodes
    .filter((n) => n.parent)
    .map((n) => ({
      id: `e-${n.parent}-${n.id}`,
      source: n.parent!,
      target: n.id,
      style: { stroke: '#94a3b8' },
    }));

  return { nodes, edges };
}

export default function MindMap({ data }: { data: string }) {
  const parsed = useMemo(() => {
    try {
      return JSON.parse(data) as MindMapData;
    } catch {
      return null;
    }
  }, [data]);

  const { nodes: initialNodes, edges: initialEdges } = useMemo(
    () => (parsed ? buildGraph(parsed) : { nodes: [], edges: [] }),
    [parsed]
  );

  const [nodes, , onNodesChange] = useNodesState(initialNodes);
  const [edges, , onEdgesChange] = useEdgesState(initialEdges);

  if (!parsed) {
    return <div className="text-red-500 text-sm p-4">Failed to parse mind map data</div>;
  }

  return (
    <div style={{ height: '400px', width: '100%' }}>
      <ReactFlow
        nodes={nodes}
        edges={edges}
        onNodesChange={onNodesChange}
        onEdgesChange={onEdgesChange}
        fitView
      >
        <Background />
        <Controls />
      </ReactFlow>
    </div>
  );
}
```

**Step 4: Create `backend/services/excel_service.py`**

```python
import duckdb
import pandas as pd
import uuid
from pathlib import Path
import logging

logger = logging.getLogger(__name__)

DUCKDB_DIR = Path("/app/uploads/duckdb")


def get_duckdb_path(source_id: uuid.UUID) -> Path:
    DUCKDB_DIR.mkdir(parents=True, exist_ok=True)
    return DUCKDB_DIR / f"{source_id}.duckdb"


async def ingest_excel(source_id: uuid.UUID, file_path: str) -> str:
    """Load Excel/CSV into DuckDB, return the DuckDB file path."""
    duckdb_path = get_duckdb_path(source_id)

    # Read with pandas
    path = Path(file_path)
    if path.suffix.lower() in ('.xlsx', '.xls'):
        df = pd.read_excel(file_path)
    else:  # .csv
        df = pd.read_csv(file_path)

    # Sanitize column names
    df.columns = [c.replace(' ', '_').replace('-', '_').lower() for c in df.columns]

    con = duckdb.connect(str(duckdb_path))
    con.execute("DROP TABLE IF EXISTS data")
    con.execute("CREATE TABLE data AS SELECT * FROM df")
    row_count = con.execute("SELECT COUNT(*) FROM data").fetchone()[0]
    schema = con.execute("DESCRIBE data").fetchdf().to_string()
    con.close()

    logger.info(f"Ingested {row_count} rows into {duckdb_path}. Schema: {schema}")
    return str(duckdb_path)


def query_excel(duckdb_path: str, sql: str) -> str:
    """Execute SQL against DuckDB and return formatted result."""
    con = duckdb.connect(duckdb_path, read_only=True)
    try:
        result = con.execute(sql).fetchdf()
        if result.empty:
            return "Query returned no results."
        return result.to_markdown(index=False)
    finally:
        con.close()


def get_table_schema(duckdb_path: str) -> str:
    """Return schema description for the data table."""
    con = duckdb.connect(duckdb_path, read_only=True)
    try:
        schema_df = con.execute("DESCRIBE data").fetchdf()
        return schema_df.to_string(index=False)
    finally:
        con.close()
```

ROUTER_PROMPT = """You are a query router. Decide if the following question should be answered using:

{schema}

Question: {question}"""


async def route_query(question: str, schema: str | None = None) -> str:
    """Returns 'sql' or 'rag'."""
    if not schema:
        return "rag"

    prompt = ROUTER_PROMPT.format(
        schema=schema or "No schema available",
        question=question
    )

    result = await qwen_client.generate(
        [{"role": "user", "content": prompt}],
        model="qwen-turbo",  # Fast, cheap model for routing
        max_tokens=10,
    )

    decision = result.strip().lower()
    return "sql" if "sql" in decision else "rag"
```

TTS_ENDPOINT = "https://nls-gateway-cn-shanghai.aliyuncs.com/stream/v1/tts"


async def text_to_speech(text: str, voice: str, appkey: str, token: str) -> bytes:
    """Call Alibaba Cloud TTS and return MP3 bytes."""
    params = {
        "appkey": appkey,
        "token": token,
        "text": text,
        "format": "mp3",
        "voice": voice,
        "sample_rate": 16000,
    }
    async with httpx.AsyncClient(timeout=60) as client:
        response = await client.get(TTS_ENDPOINT, params=params)
        if response.status_code != 200:
            raise Exception(f"TTS failed: {response.status_code} {response.text[:200]}")
        return response.content

Format: alternating HOST: and GUEST: lines.
The conversation should be engaging, informative, and 8-12 exchanges.

import tempfile
import os
from pydub import AudioSegment
from backend.services.tts_client import text_to_speech
from backend.core.config import settings

@router.post("/{notebook_id}/studio/podcast")
async def generate_podcast(
    notebook_id: uuid.UUID,
    db: AsyncSession = Depends(get_db),
    current_user: User = Depends(get_current_user),
):
    await permission_service.check_permission(db, notebook_id, current_user.id, "viewer")

    if not settings.ALIBABA_TTS_APPKEY or not settings.ALIBABA_TTS_TOKEN:
        raise HTTPException(status_code=501, detail="TTS not configured")

    context = await _get_source_context(db, notebook_id)
    if not context:
        raise HTTPException(status_code=400, detail="No source documents available")

    # Generate dialogue
    prompt = PODCAST_PROMPT.format(context=context[:6000])
    dialogue = await qwen_client.generate(
        [{"role": "system", "content": "Generate engaging podcast dialogue."},
         {"role": "user", "content": prompt}]
    )

    # Parse dialogue lines
    lines = []
    for line in dialogue.strip().split('\n'):
        line = line.strip()
        if line.startswith('HOST:'):
            lines.append(('host', line[5:].strip()))
        elif line.startswith('GUEST:'):
            lines.append(('guest', line[6:].strip()))

    if not lines:
        raise HTTPException(status_code=500, detail="Failed to generate dialogue")

    # Generate audio for each line
    host_voice = "ailun"   # Female voice
    guest_voice = "aicheng"  # Male voice

    segments = []
    with tempfile.TemporaryDirectory() as tmpdir:
        for i, (speaker, text) in enumerate(lines):
            voice = host_voice if speaker == 'host' else guest_voice
            audio_bytes = await text_to_speech(
                text, voice,
                settings.ALIBABA_TTS_APPKEY,
                settings.ALIBABA_TTS_TOKEN
            )
            tmp_path = os.path.join(tmpdir, f"seg_{i}.mp3")
            with open(tmp_path, 'wb') as f:
                f.write(audio_bytes)
            segments.append(AudioSegment.from_mp3(tmp_path))
            # 300ms pause between segments
            segments.append(AudioSegment.silent(duration=300))

        # Merge all segments
        combined = segments[0]
        for seg in segments[1:]:
            combined += seg

        out_path = os.path.join(tmpdir, "podcast.mp3")
        combined.export(out_path, format="mp3")

        with open(out_path, 'rb') as f:
            mp3_bytes = f.read()

    return StreamingResponse(
        io.BytesIO(mp3_bytes),
        media_type="audio/mpeg",
        headers={"Content-Disposition": 'attachment; filename="podcast.mp3"'}
    )
```
